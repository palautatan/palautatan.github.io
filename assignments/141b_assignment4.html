# SIMPLE DATA BASE FUNCTION
def ndb_search(this_term, key):
    """
    Input: A search term (string) and a API key (string)
    Output: A list of query results
    Performs a standard search query with no particular food group specified
    """
    base_url = "https://api.nal.usda.gov/ndb/search/"
    key_term = "&api_key=" + key
    final_url = base_url + "?format=json&q=" + this_term + key_term
    #print(final_url)
    request = requests.get(final_url)
    request_json = request.json()
    try:
        return request_json["list"]["item"]
    except:
        #return 0
        return([np.NaN])

# REVISED NDB_SEARCH FUNCTION
def ndb_search2(row, df, key):
    """
    Input: A row index (Pandas dataframe index), a dataframe (Pandas dataframe), and an API key (string)
    Output: A list of query results
    Performs a specialized search for foods in fresh   
    """
    # PARSE DATAFRAME FOR INFORMATION
    this_row = df.iloc[row]
    this_term = (this_row[5]).replace("_", " ") # name
    fg = this_row[6] # fruit or vegetable
    
    # CONSTRUCT QUERY CALL
    base_url = "https://api.nal.usda.gov/ndb/search/"
    key_term = "&api_key=" + key
    if fg == "fruit":
        final_url = base_url + "?format=json&q=" + this_term + "&ds=Standard Reference" +"&fg=Fruits and Fruit Juices" + key_term
    else:
        final_url = base_url + "?format=json&q=" + this_term + "&ds=Standard Reference" + "&fg=Vegetables and Vegetable Products" + key_term
    request = requests.get(final_url)
    request_json = request.json()
    
    # CATCH FOR ERRORS
    try:
        return request_json["list"]["item"]
    except:
        return([np.NaN])

def ndb_report(ndb_no, key):
    """
    Input: NDB number (string) and API Key (String)
    Output: Nutrition report  from USDA food composition database (List of dictionaries)
    Returns basic nutrition report of desired food item
    """
    final_url = "https://api.nal.usda.gov/ndb/V2/reports?ndbno="+str(ndb_no)+"&type=b&format=json&api_key="+key
    request = requests.get(final_url)
    request_json = request.json()
    return(request_json["foods"][0]["food"]["nutrients"])
